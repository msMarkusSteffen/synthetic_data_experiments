\begin{frame}
    \frametitle{Was sind synthetische Daten}
    %\small \textit{Ein synthetischer Datensatz besitzt die gleichen mathematischen Eigenschaften wie die realen Daten, die er ersetzt, enthält jedoch keine der ursprünglichen Informationen. Er wird erzeugt, indem man echte Daten nimmt, ein generatives Machine-Learning-Modell darauf trainiert und daraus einen neuen Datensatz generiert. Das Ergebnis spiegelt die allgemeinen Muster des Originals wider, liefert aber genug ‚Rauschen‘, um die echten Daten genügend zu maskieren.} frei \parencite{mit_sloan_what_2023}
    \begin{itemize}
        \item Ein synthetischer Datensatz hat: \parencite{mit_sloan_what_2023}
        \begin{itemize}
            \item die gleichen math. Eigenschaften wie realer Datensatz
            \item enthält keine ursprünglichen Informationen
            \item wird mittels generativer Lernmethoden erzeugt (Training)
            \item genug Rauschen um echte Daten genügend zu maskieren
        \end{itemize}
    \end{itemize}
    \medskip
    \begin{itemize}        
        \item Synthetische Daten basieren auf 4 Grundpfeilern % Ressource 
        \begin{itemize}
            \item Fidelity
            \item Utility
            \item Privacy
            \item Reproducebility
        \end{itemize}
    \end{itemize}
    
\end{frame}

\begin{frame}
    \frametitle{Einsatzzwecke für synthetische Daten, oder : wozu braucht man das?}
    \begin{itemize}
        \item Datenschutz und Anonymsierung
        \item Datenverfügbarkeit
        \item Rapid Prototyping 
        \item Kostenreduktion %Vermeidung teurer Datenerhebung (aufwändige Sensoraufnahmen, manuelle Annotation)    
        \item Data Augementation    
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Herausforderungen}
    \begin{itemize}
        \item Datenqualität
        \item Trade-Off Qualität vs Privacy
        \item Bewertung
        \begin{itemize}
            \item Wie Ergebnisse bewerten ? - keine einheitlicher Standard für Metrik (Vergleichbarkeit) \parencite{escudie_evaluating_2026}
            \item Wie verschiedene Daten vergleichen % Modell immer zugeschnitten, Teilen von Daten schwierig
            \item Wie verschiedene Methodiken vergleichen ? % Hyperparametertuning, computational Time
            \item Wie Anonymität einschätzen ? %epsilon Delta 
        \end{itemize}
        \item sauberes Anforderungsmanagement % Welchen Anforderungen muss der synth Datensatz standhalten, z.B. Pipeline -> Dashboard, Fachliche Anforderungen auf Dashboard ?
        \item Technische Umsetzung 
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Training und Erzeugung}
        \begin{itemize}
            \item Training
            \begin{itemize}
                \item Startpunkt vorprozessierte reale Daten (analog CTGAN)
                \item Verrauschung durch Noise Sheduler 
                \item Sinosuidales Embedding (quasi t dem Netz mitgeben)
                \item Netz lernt Realdaten für Schritt t aus Rauschen zu denoisen
                \item Metrik : Kullback-Leibler Divergenz + 
                \item Essenz der Informationen im Rauschen enthalten
            \end{itemize}
            \item Generierung
            \begin{itemize}
                \item Erzeugung latenter Tensor (selbe Größe wie vorprozessiertes Datenset)
                \item Revers Prozess durch Abschalen der Rauschschichten
                \item Umkehrung der Preprocessng Schritte (scaling, one-hot encoding)
            \end{itemize}
        \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Verrauschung und Embedding}
        \begin{itemize}
            \item Verrauschung mittels noise Sheduler \parencite{truda_generating_2023}
            \begin{itemize}
                \item jedes Element im aktuellen Batch bekommt noise aufaddiert
                \item $x_{t+1} = x_t + z_t$ mit $z_t \sim \mathcal{N} \left(0,1\right)\sqrt{\beta_t}$, wobei $\beta_t = \frac{t}{T}$
            \end{itemize}
            \item Sinusoidales Embedding des Noiser Steps \parencite{vaswani_attention_2023}
            \begin{itemize}
                \item Ziel Netz soll Verrauschungsstufe mitbekommen (embedd vs concat)
                \item Vector mit Sinus und Kosinus Einträgen, wobei jede Dimension für ein festes t einzigartig ist
            \end{itemize}
        \end{itemize}

\begin{frame}
    \frametitle{Zusammenfassung}
    \begin{itemize}
        \item Synthetische Daten Mimiken reale Daten
        \item Es gibt keine einheitlich akzeptierte Metrik (Bewertungskriterien)
        \item Use Case sollte klar und auch umfassend definiert werden (Endanwendung, ETL, ..)
        \item Viele Wege (Methodiken) führen nach Rom (zum Ergebnis)
        \item CTGAN: funktioniert wie zwei Gegenspieler, wobei Generator aus Rauschen Fake Daten erzeugt und versucht Diskriminator auszutricksen
        \item Table Diffusion: neuronales Netz lernt wie Daten verrauscht werden und kann aus Rauschen synthetische Daten rekreiern.
    \end{itemize}
\end{frame}
